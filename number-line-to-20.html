20. Write a program that asks the user to input 2 lists of numbers, 5 numbers per list. Your program should identify the common numbers in the lists or the numbers that are present in both lists. See sample lists and output below. (C++) Input: Line 1: 20 3 59 1 3 Line 2: 1 20 1 4 3 Output: 20 3 1 Input: Line 1: 3 5 86 31 23 Line 2: 21 54 87 13 7 Output: None


#include <iostream>

#include <algorithm>

#include <unordered_set>

#include <vector>

int main() {

   std::vector<int> list1, list2;

   int number;

   // Input for the first list

   std::cout << "Line 1:";

   for (int i = 0; i < 5; i++) {

       std::cin >> number;

       list1.push_back(number);

   }

   // Input for the second list

   std::cout << "Line 2:";

   for (int i = 0; i < 5; i++) {

       std::cin >> number;

       list2.push_back(number);

   }

   // Use an unordered set to keep track of the common numbers

   std::unordered_set<int> commonNumbers;

   for (int number : list1) {

       if (std::find(list2.begin(), list2.end(), number) != list2.end()) {

           commonNumbers.insert(number);

       }

   }

   // Output the common numbers

   if (commonNumbers.size() > 0) {

       std::cout << "Output: ";

       for (int number : commonNumbers) {

           std::cout << number << " ";

       }

   } else {

       std::cout << "Output: None";

   }

   return 0;

}
