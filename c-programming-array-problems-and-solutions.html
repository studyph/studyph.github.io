#include <iostream>

using namespace std;

int main() {

  int n, i;

  cout << "Enter the length of the array: ";

  cin >> n;

  int arr[n];

 

  // Accept numbers from user and validate them

  cout << "Enter " << n << " numbers between -100 to 100:\n";

  for (i = 0; i < n; i++) {

     cin >> arr[i];

     if (arr[i] < -100 || arr[i] > 100) {

        cout << "Invalid input! Number must be between -100 to 100.\n";

        i--; // repeat this iteration

     }

  }

 

  // Ask user for operator

  char op;

  cout << "Enter an operator (+, -, *, /): ";

  cin >> op;

 

  // Calculate the result based on operator

  int result = arr[0];

  for (i = 1; i < n; i++) {

     switch(op) {

        case '+':

           result += arr[i];

           break;

        case '-':

           result -= arr[i];

           break;

        case '*':

           result *= arr[i];

           break;

        case '/':

           if (arr[i] == 0) {

              cout << "Cannot divide by zero!\n";

              return 0;

           }

           result /= arr[i];

           break;

        default:

           cout << "Invalid operator!\n";

           return 0;

     }

  }

 

  // Output the result

  cout << "The result is " << result << endl;

  return 0;

}

8. C++ Programming. Task : This program takes (n) number of elements from the user and stores it in the (arr) array. Find the 3 maximum elements in that array. (use pointers) Plase help me...


Answer:

Here's a solution that makes use of pointers to find the three largest elements in an array.

#include <iostream>

#include <algorithm>

using namespace std;

const int N = 1e5;

int main() {

   int n;

   cout << "Enter the number of elements: ";

   cin >> n;

   int arr[N];

   cout << "Enter the elements: ";

   for (int i = 0; i < n; i++) {

       cin >> arr[i];

   }

   int* max1 = max_element(arr, arr + n);

   int* max2 = max1;

   int* max3 = max1;

   for (int* i = arr; i != arr + n; i++) {

       if (*i > *max3) {

           max3 = i;

           if (*max3 > *max2) {

               swap(max2, max3);

               if (*max2 > *max1) {

                   swap(max1, max2);

               }

           }

       }

   }

   cout << "The 3 maximum elements are: " << *max1 << " " << *max2 << " " << *max3 << endl;

   return 0;

}

Explanation:

First, we use the algorithm library's max element function to find the pointer to the array's maximum element.

Following that, we initialize three pointers max1, max2, and max3 to the maximum element's pointer.

Then, using a pointer, we traverse the array and update the values of max1, max2, and max3 as needed. We compare each element in the array to max3. If it exceeds max3, we update max3 to point to it. Then we compare max3 to max2, and if max3 is greater, the values of max2 and max3 are swapped. Finally, we compare max2 to max1, and if max2 is greater, the values of max1 and max2 are swapped.

Finally, we output the three most important elements.
