AB = 5.5x - 3.7

= 5.5 (7.5) - 3.7

= 41.25 - 3.7

AB = 37.55

BC = 4.3x + 5.3

= 4.3 (7.5) + 5.3

= 32.25 + 5.3

BC = 37.55
